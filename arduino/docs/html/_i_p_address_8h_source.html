40  // Access the raw byte array containing the address. Because this returns a pointer
41  // to the internal structure rather than a copy of the address this function should only
42  // be used when you know that the usage of the returned uint8_t* will be transient and not
43  // stored.
44  uint8_t* raw_address() { return _address.a8; }

46 public:
47  // Constructors

50  IPAddress(uint32_t address);

52 
53  // Overloaded cast operator to allow IPAddress objects to be used where a pointer
54  // to a four-byte uint8_t array is expected
55  operator uint32_t() { return _address.a32; }
56  bool operator==(const IPAddress& addr) { return _address.a32 == addr._address.a32; };
57  bool operator==(const uint8_t* addr);

59  // Overloaded index operator to allow getting and setting individual octets of the address
